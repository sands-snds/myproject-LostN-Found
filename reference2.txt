<script>

const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbxTTu4g6Dr5iqwunn0zzNxdUtsPt7jvmcA4d6chFju7PXoRECfq2o0qPrgarUOWIcH3SA/exec";

// Load saved count on page load

let billCount = Number(localStorage.getItem("billCount")) || 0;
document.getElementById("counter").textContent = "Bills generated: " + billCount;

function generateBill(){

let name=document.getElementById("cusName").value;
let consumpt=Number(document.getElementById("consumption").value);
let status=document.getElementById("status").value;

// Validation

if(name=="" || document.getElementById("consumption").value=="" || status==""){
    alert("Complete all fields.");
    return;
}

if(isNaN(consumpt) || consumpt<1){
    alert("Please input a valid number.");
    return;
}

// Discount Rate
let discount = 0;
switch(status){
    case "Regular":
        discount = 0;
        break;
    case "SeniorCitizen":
        discount = 0.25;
        break;
    case "SParent":
        discount = 0.15;
        break;
    default:
        discount = 0;
}

// Cubic Meter Rate
let rCubicMeter = 0;

if (consumpt >= 1 && consumpt <= 20){
    rCubicMeter = 25;
} else if (consumpt >= 21 && consumpt <= 40){
    rCubicMeter = 35;
} else if (consumpt >= 41 && consumpt <= 60){
    rCubicMeter = 45;
} else if (consumpt >= 61){
    rCubicMeter = 60;
} else {
    alert("Please input a valid number.");
    return;
}

// Calculation
let total = consumpt * rCubicMeter;
let dTotal = total - (total * discount);

// querySelector()

document.querySelector("#title").textContent="Water Bill Successfully Calculated";

// querySelectorAll()

let inputs=document.querySelectorAll("input");

inputs.forEach(function(box){
    box.style.background="#fff8dd";
});

// getElementsByClassName()

let result=document.getElementsByClassName("result")[0];
result.innerHTML="";

// createElement()

let heading=document.createElement("h2");

heading.textContent="Customer Bill";

heading.setAttribute("class","highlight");

result.appendChild(heading);

// create Table

let table=document.createElement("table");

table.innerHTML=`

<tr>
<th>Description</th>
<th>Value</th>
</tr>

<tr>
<td>Name</td>
<td>${name}</td>
</tr>

<tr>
<td>Water Consumption</td>
<td>${consumpt}</td>
</tr>

<tr>
<td>Status</td>
<td>${status}</td>
</tr>

<tr>
<td>Cubic Rate</td>
<td>₱${Number(rCubicMeter).toFixed(2)}</td>
</tr>

<tr>
<td>Discount</td>
<td>${(discount*100).toFixed(0)}%</td>
</tr>

<tr>
<td>Total</td>
<td>₱${total.toFixed(2)}</td>
</tr>

<tr>
<td>Discounted Total</td>
<td>₱${dTotal.toFixed(2)}</td>
</tr>

`;

result.appendChild(table);

// getElementsByTagName()

let cells=document.getElementsByTagName("tr");

for(let i=0;i<cells.length;i++){

    if(i%2==1){
        cells[i].style.fontWeight="bold";
    }

}

// removeChild()

let oldParagraph=document.querySelector(".note");

if(oldParagraph!=null){
    result.removeChild(oldParagraph);
}

// createElement()

let note=document.createElement("p");

note.textContent="Saving to Google Sheet...";

note.setAttribute("class","note");

result.appendChild(note);

// data to Google Sheet

let genBtn=document.getElementById("genBtn");
genBtn.disabled=false;

fetch(WEB_APP_URL, {
    method: "POST",
    mode: "no-cors",
    headers: {
        "Content-Type": "text/plain"
    },
    body: JSON.stringify({
        cusName: name,
        consumption: consumpt,
        status: status,
        cCubicRate: rCubicMeter,
        discount: discount,
        total: total,
        dTotal: dTotal
    })
})
.then(function(){
    note.textContent="Bill generated and saved to Google Sheet.";
    genBtn.disabled=false;
})
.catch(function(err){
    note.textContent="Bill generated, but saving to Google Sheet failed.";
    genBtn.disabled=false;
});

// Update counter

billCount = billCount + 1;
localStorage.setItem("billCount", billCount);
document.getElementById("counter").textContent = "Bills generated: " + billCount;

// Clear fields

document.getElementById("cusName").value="";
document.getElementById("consumption").value="";
document.getElementById("status").value="Regular";

}

</script>
